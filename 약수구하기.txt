#약수구하기. 코드업에서는 원래 풀던대로 하면 시간 초과가 났다! 제곱근으로 구하는 방법이 있었다!

import math

n = int(input())

# 16
# 1 2 4 8 16


sum_val = 0

for i in range(1, int(math.sqrt(float(n)))+1):
    if n % i == 0:
        if n // i == i:
            sum_val += i
        else:
            sum_val = sum_val + i + n // i
        
print(sum_val)
